TASK 01:
INCLUDE Irvine32.inc

.data
var WORD ?

.code
main proc

MOV EBX, 5h ; MOVE does not exist
ADD CX, 5 ; INC cannot take an immediate value
MOV var, AX ; Cannot move into an immediate like 10h
MOV DX, AX ; Operand size mismatch
ADD BX, 5 ; Destination cannot be an immediate
MOV EAX, 45d ; Missing comma between operands
MOV AX, 5 ; Both operands cannot be immediates
ADD AX, 5 
call DumpRegs

exit
main endp
end main
